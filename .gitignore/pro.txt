Structure du projet 

/gestion_dechets
│── /public               # Dossier accessible par le navigateur (CSS, JS, images)
│   ├── /css              # Fichiers CSS
│   ├── /js               # Fichiers JavaScript
│   ├── /images           # Images et icônes
│   ├── index.php         # Page d'accueil
│   ├── login.php         # Page de connexion
│   ├── register.php      # Page d'inscription
│   ├── dashboard.php     # Tableau de bord après connexion
│── /config               # Connexion à la base de données
│   ├── database.php      # Fichier de connexion MySQL
│   ├── config.php        # Paramètres globaux
│── /models               # Gestion des données (CRUD pour utilisateurs, collectes, etc.)
│   ├── User.php          # Modèle utilisateur
│   ├── Collecte.php      # Modèle collecte de déchets
│   ├── Entreprise.php    # Modèle entreprise de collecte
│── /controllers          # Traitement des requêtes et logique métier
│   ├── AuthController.php  # Inscription, connexion, déconnexion
│   ├── CollecteController.php # Gestion des collectes
│   ├── MessageController.php  # Gestion des messages entre utilisateurs
│── /views                # Interfaces utilisateur (HTML + PHP)
│   ├── /layouts          # Header, footer
│   ├── /collectes        # Pages de suivi des collectes
│   ├── /entreprises      # Liste des entreprises de collecte
│── /routes               # Définition des routes PHP
│   ├── web.php           # Fichier de gestion des routes
│── /uploads              # Stockage des images et documents
│── .htaccess             # Fichier Apache (réécriture d’URL)
│── README.md
